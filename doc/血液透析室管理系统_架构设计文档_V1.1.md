### 1.2 处方 → 医嘱 → 透析录入联动链路

```
医生开具透析处方                医生开具长期医嘱
      │                              │
      ▼                              ▼
 prescriptions表              long_term_orders表
 （透析参数：频次/时长/        （用药明细：药品名/剂量/
   透析器/抗凝/干体重）          用法/执行频次/有效期）
      │                              │
      └──────────────┬───────────────┘
                     │ 护士开始录入透析记录时
                     ▼ 系统自动调用（OrderAutoFill服务）
              透析录入页面自动带入：
              ① 透析参数（来自处方）
              ② 本次应执行的用药医嘱列表
              ③ 护士逐条确认执行 → 写入 order_executions表
```

## 三、数据库设计

### 3.1 数据库总览

```
数据库：hemodialysis_db（PostgreSQL 16，UTF-8编码）

核心业务表（22张）：
  用户与权限：users
  患者管理：patients
  处方与医嘱：prescriptions, long_term_orders, order_executions  ← 新增2张
  透析业务：dialysis_records, vital_signs, complications
  检验管理：lab_results
  血管通路：vascular_accesses, thrombolysis_records, cvc_risk_assessments
  传染病：infection_screenings, infection_monitoring
  排班：schedules, machines
  设备：devices, water_quality_records, consumables, consumable_stocks
  质量管理：cqi_records, defect_reports, qc_reports
  系统：audit_logs, alerts, system_configs

核心联动关系：
  prescriptions ──(1:N)──> long_term_orders
  long_term_orders ──(1:N)──> order_executions
  order_executions ──(N:1)──> dialysis_records
```

#### 患者档案表 patients

```sql
CREATE TABLE patients (
  id                    UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name                  VARCHAR(50) NOT NULL,
  gender                CHAR(1) NOT NULL CHECK (gender IN ('M','F')),
  dob                   DATE NOT NULL,
  id_card_encrypted     TEXT,
  phone_encrypted       TEXT,
  family_contact        JSONB,
  address               TEXT,
  primary_diagnosis     VARCHAR(100) NOT NULL,
  ckd_stage             SMALLINT CHECK (ckd_stage BETWEEN 1 AND 5),
  comorbidities         TEXT[],
  dialysis_start_date   DATE NOT NULL,
  dialysis_mode         VARCHAR(20) NOT NULL,
  status                VARCHAR(20) NOT NULL DEFAULT 'active',
  status_note           TEXT,
  status_changed_at     DATE,
  isolation_zone        VARCHAR(20) DEFAULT 'normal',
  consent_dialysis      BOOLEAN DEFAULT false,
  consent_dialysis_date DATE,
  consent_reuse         BOOLEAN DEFAULT false,
  consent_reuse_date    DATE,
  created_by            UUID REFERENCES users(id),
  created_at            TIMESTAMP DEFAULT NOW(),
  updated_at            TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_patients_status ON patients(status);
CREATE INDEX idx_patients_name   ON patients(name);
CREATE INDEX idx_patients_iso    ON patients(isolation_zone);
```

#### 透析处方表 prescriptions

```sql
CREATE TABLE prescriptions (
  id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  patient_id          UUID NOT NULL REFERENCES patients(id),

  -- 透析参数
  frequency_per_week  SMALLINT NOT NULL,
  duration_hours      NUMERIC(3,1) NOT NULL,
  dialyzer_model      VARCHAR(100),
  dialyzer_area       NUMERIC(4,2),
  dialyzer_flux       VARCHAR(10),          -- high/low（高通量/低通量）
```

#### 长期医嘱表 long_term_orders ★核心新增

```sql
-- 医嘱类型说明：
```

**保存处方时的后端联动逻辑：**

```javascript
// src/routes/prescriptions.js（节选）
router.post('/:patientId', auth, rbac(['doctor','admin']), async (req, res) => {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 1. 将旧处方归档
    await client.query(
      `UPDATE prescriptions SET is_current=false, valid_until=CURRENT_DATE
       WHERE patient_id=$1 AND is_current=true`, [req.params.patientId]
    );

    // 2. 插入新处方
    const { rows: [newRx] } = await client.query(
      `INSERT INTO prescriptions (patient_id, ..., prescribed_by)
       VALUES ($1, ..., $2) RETURNING *`,
      [req.params.patientId, ..., req.user.id]
    );

    // 3. ★ 联动更新：同步抗凝医嘱（如果已有肝素医嘱则更新剂量）
    const { anticoagulant, heparin_prime_dose, heparin_maintain } = req.body;
    if (anticoagulant !== 'none') {
      await client.query(
        `UPDATE long_term_orders
         SET dose = $1, notes = '处方同步更新', updated_at = NOW()
         WHERE patient_id = $2
           AND status = 'active'
           AND order_type = 'dialysis_drug'
           AND drug_name ILIKE '%肝素%'`,
        [`首剂${heparin_prime_dose}IU，维持${heparin_maintain}IU/h`,
         req.params.patientId]
      );
    }

    // 4. 将新处方ID关联到活跃医嘱
    await client.query(
      `UPDATE long_term_orders SET prescription_id=$1
       WHERE patient_id=$2 AND status='active'`,
      [newRx.id, req.params.patientId]
    );

    await client.query('COMMIT');
    res.json({ code: 200, data: newRx, message: '处方开具成功' });
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally { client.release(); }
});
```

### 6.1 长期医嘱自动带入服务（OrderAutoFill）★核心新增

```javascript
// src/services/OrderAutoFill.js

class OrderAutoFill {
  constructor(db) { this.db = db; }

  /**
   * 护士开始录入透析记录时调用，返回当次处方参数 + 应执行的医嘱列表
   * GET /api/dialysis/prepare?patientId=&date=
   */
  async prepareForDialysis(patientId, sessionDate) {
    // 1. 获取当前有效处方
    const { rows: [prescription] } = await this.db.query(
      `SELECT * FROM prescriptions
       WHERE patient_id = $1 AND is_current = true
       LIMIT 1`, [patientId]
    );

    // 2. 获取所有有效医嘱
    const { rows: allOrders } = await this.db.query(
      `SELECT * FROM long_term_orders
       WHERE patient_id = $1
         AND status = 'active'
         AND valid_from <= $2
         AND (valid_until IS NULL OR valid_until >= $2)
       ORDER BY order_type, created_at`,
      [patientId, sessionDate]
    );

    // 3. 判断每条医嘱今日是否应执行
    const ordersToday = [];
    for (const order of allOrders) {
      const shouldExecute = await this.shouldExecuteToday(
        order, patientId, sessionDate
      );
      ordersToday.push({ ...order, shouldExecute, ...shouldExecute });
    }

    return { prescription, ordersToday };
  }

  /**
   * 判断某条医嘱在指定透析日是否应执行
   * 返回 { required: bool, reason: string, lastExecution: date }
   */
  async shouldExecuteToday(order, patientId, sessionDate) {
    // 获取最近一次执行记录
    const { rows: [lastExec] } = await this.db.query(
      `SELECT execution_date, status FROM order_executions
       WHERE long_term_order_id = $1
         AND status != 'skipped'
       ORDER BY execution_date DESC LIMIT 1`,
      [order.id]
    );

    const last = lastExec?.execution_date;
    const today = new Date(sessionDate);

    switch (order.frequency) {
      case 'every_session':   // 每次透析必须执行
        return { required: true, reason: '每次透析执行', lastExecution: last };

      case 'qd':              // 每日一次，若今日已执行则不再显示
        const todayExec = await this.db.query(
          `SELECT id FROM order_executions
           WHERE long_term_order_id=$1 AND execution_date=$2
             AND status='executed' LIMIT 1`,
          [order.id, sessionDate]
        );
        return {
          required: todayExec.rows.length === 0,
          reason: '每日一次',
          lastExecution: last
        };

      case 'tiw':             // 每周三次（透析日），始终执行
        return { required: true, reason: '每周三次透析日执行', lastExecution: last };

      case 'qw':              // 每周一次
        if (!last) return { required: true, reason: '首次执行', lastExecution: null };
        const daysSinceLast = Math.floor(
          (today - new Date(last)) / 86400000
        );
        return {
          required: daysSinceLast >= 7,
          reason: daysSinceLast >= 7
            ? `上次${last}，本周应执行`
            : `上次${last}，本周已执行`,
          lastExecution: last
        };

      case 'q2w':             // 每两周一次
        if (!last) return { required: true, reason: '首次执行', lastExecution: null };
        const days2 = Math.floor((today - new Date(last)) / 86400000);
        return {
          required: days2 >= 14,
          reason: days2 >= 14 ? `距上次已${days2}天，应执行` : `下次应在${new Date(new Date(last).getTime() + 14*86400000).toLocaleDateString('zh-CN')}`,
          lastExecution: last
        };

      case 'qm':              // 每月一次
        if (!last) return { required: true, reason: '首次执行', lastExecution: null };
        const daysMon = Math.floor((today - new Date(last)) / 86400000);
        return {
          required: daysMon >= 28,
          reason: daysMon >= 28 ? '本月应执行' : `上次${last}，本月已执行`,
          lastExecution: last
        };

      default:
        return { required: false, reason: '请手动判断', lastExecution: last };
    }
  }
}

module.exports = OrderAutoFill;
```

```javascript
// src/services/KtvCalculator.js
class KtvCalculator {
  // Daugirdas II 公式
  static calculate(preBUN, postBUN, duration, ufVolume, postWeight) {
    if (!preBUN || !postBUN || preBUN <= 0) {
      return { ktv: null, urr: null, isAdequate: null };
    }
    const R = postBUN / preBUN;
    const ktv = -Math.log(R - 0.008 * duration) + (4 - 3.5 * R) * (ufVolume / postWeight);
    const urr = (1 - R) * 100;
    return {
      ktv: Math.round(ktv * 100) / 100,
      urr: Math.round(urr * 10) / 10,
      isAdequate: ktv >= 1.2 && urr >= 65
    };
  }

  static calcUFVolume(preWeight, postWeight, dryWeight) {
    const ufVolume = (preWeight - postWeight) * 1000;
    const ufPct = ((preWeight - dryWeight) / dryWeight) * 100;
    return {
      ufVolume: Math.round(ufVolume),
      ufPct: Math.round(ufPct * 10) / 10,
      isOverLimit: ufPct > 5
    };
  }
}
```

```javascript
      FROM (
        SELECT session_date,
          COUNT(*) AS pc, COUNT(DISTINCT nurse_id) AS nc,
          BOOL_OR(is_circuit_clotted) AS is_circuit_clotted,
          BOOL_OR(is_membrane_ruptured) AS is_membrane_ruptured,
          BOOL_OR(is_avf_session) AS is_avf_session
        FROM dialysis_records
        WHERE EXTRACT(YEAR FROM session_date)=$1
          AND EXTRACT(MONTH FROM session_date)=$2
        GROUP BY session_date
      ) d`, [year, month]);

    const { rows: [inj] } = await this.db.query(
      `SELECT COUNT(*) AS cnt FROM complications c
       JOIN dialysis_records dr ON c.dialysis_record_id=dr.id
       WHERE EXTRACT(YEAR FROM dr.session_date)=$1
         AND EXTRACT(MONTH FROM dr.session_date)=$2
         AND c.type='avf_injury'`, [year, month]);

      `SELECT SUM(catheter_days) AS days, COUNT(*) FILTER(WHERE status='infected') AS cnt
       FROM infection_monitoring WHERE monitor_year=$1 AND monitor_month=$2`, [year, month]);

    const total = parseInt(s.total_sessions);
    const avf   = parseInt(s.avf_sessions);
    const ratio = parseFloat(s.total_patient_sessions) / parseFloat(s.total_nurse_sessions);
```

```javascript
exports.decrypt = (s) => {
  const [ivH, tagH, dataH] = s.split(':');
  const d = crypto.createDecipheriv('aes-256-gcm', KEY, Buffer.from(ivH,'hex'));
  d.setAuthTag(Buffer.from(tagH,'hex'));
  return Buffer.concat([d.update(Buffer.from(dataH,'hex')), d.final()]).toString('utf8');
};
```

```javascript
// ✅ 参数化查询（防SQL注入）
const { rows } = await pool.query(
  'SELECT * FROM long_term_orders WHERE patient_id=$1 AND status=$2',
  [patientId, 'active']

const client = await pool.connect();
try {
  await client.query('BEGIN');
  // 1. 归档旧处方
  await client.query(`UPDATE prescriptions SET is_current=false WHERE patient_id=$1 AND is_current=true`, [pid]);
  // 2. 插入新处方
  const { rows: [rx] } = await client.query(`INSERT INTO prescriptions (...) VALUES (...) RETURNING *`, [...]);
  // 3. 关联现有医嘱到新处方
  await client.query(`UPDATE long_term_orders SET prescription_id=$1 WHERE patient_id=$2 AND status='active'`, [rx.id, pid]);
  await client.query('COMMIT');
} catch(err) {
  await client.query('ROLLBACK');
  throw err;
} finally { client.release(); }
```
